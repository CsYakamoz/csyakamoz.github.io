Node.js 调用微信小程序图片校验接口

2020-09-09

 1const fs = require("fs");
 2const request = require("request");
 3
 4request.post(
 5  {
 6    uri:
 7      "https://api.weixin.qq.com/wxa/img_sec_check?access_token=ACCESS_TOKEN",
 8
 9    formData: {
10      // 方法 1
11      media: fs.createReadStream("path_to_img"),
12
13      // 方法 2
14      attachments: [
15        // 只认下标为 0 的
16        fs.createReadStream("path_to_img"),
17
18        // 下标 1 及之后的会被微信忽略
19        ...,
20      ],
21
22      // 方法 3
23      media: {
24        value: buffer,
25        options: {
26          // 必须有 filename, 值可任意但不能为空串
27          filename: "default",
28          // contentType 可有可无，且值任意
29          contentType: "image/jpeg",
30        },
31      },
32
33      // 方法 4
34      media: request.get("uri_to_img"),
35    },
36  },
37  (error, resp, body) => {
38    if (error) {
39      console.log(error);
40      return;
41    }
42
43    console.log(body);
44  }
45);

方法 1, 2 的使用场景：文件在本地

方法 3 的使用场景：参数是图片的 base64 内容，在调用微信接口前，将其转换成 Buffer

方法 4 的使用场景：知道图片资源的 uri
