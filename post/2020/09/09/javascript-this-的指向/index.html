Javascript this 的指向

2020-09-09

 1// Notice: the environment is Node.js
 2
 3global.val = "global";
 4this.val = "exports";
 5
 6function normalFun() {
 7  console.log(this.val);
 8}
 9
10const funcExpression = function () {
11  console.log(this.val);
12};
13
14const arrowFun = () => {
15  console.log(this.val);
16};
17
18const scope = {
19  val: "obj",
20
21  a: function () {
22    console.log(this.val);
23  },
24
25  b: () => {
26    console.log(this.val);
27  },
28
29  c: function () {
30    return () => {
31      console.log(this.val);
32    };
33  },
34
35  d: function () {
36    return function () {
37      console.log(this.val);
38    };
39  },
40};
41
42const _c = scope.c;
43const _d = scope.d;
44
45normalFun(); // global
46funcExpression(); // global
47arrowFun(); // exports
48scope.a(); // obj
49scope.b(); // exports
50scope.c()(); // obj
51scope.d()(); // global
52_c()(); // global
53_d()(); // global
